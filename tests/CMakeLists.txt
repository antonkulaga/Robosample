set(OpenMM_ON OFF)
find_package(OpenMM)
if(OpenMM_FOUND)
        set(OpenMM_ON ON)
endif()

if ( OpenMM_ON )
        find_package(OpenMM REQUIRED)
        INCLUDE_DIRECTORIES(${OpenMM_INCLUDE_DIR})
        INCLUDE_DIRECTORIES(${CMAKE_SOURCE_DIR}/src)
	MESSAGE("Include OpenMM_INCLUDE_DIR: ${OpenMM_INCLUDE_DIR} ")
else( OpenMM_ON )
	MESSAGE( FATAL_ERROR "OpenMM_ON FALSE")
endif( OpenMM_ON )
